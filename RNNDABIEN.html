def create_lag_only_features(df_input):
    df = df_input.copy()

    # 1. Lag ca price (OK)
    df['Lag_1'] = df['PRICE'].shift(1)

    # 2. Relative change
    df['Price_Pct'] = df['PRICE'].pct_change().shift(1)

    df['Brent_Pct'] = df['Brent'].pct_change().shift(1)
    df['WTI_Pct'] = df['WTI'].pct_change().shift(1)
    df['USDVND_Pct'] = df['VND_USD'].pct_change().shift(1)

    # 3. Interaction
    df['Brent_WTI_Spread'] = (df['Brent'] - df['WTI']).shift(1)
    df['Brent_WTI_Ratio'] = (df['Brent'] / (df['WTI'] + 1e-8)).shift(1)

    # Bin tng tc phc tp: USD (hm qua) * Brent% (hm qua)
    df['USD_Brent_Interact'] = df['VND_USD'].shift(1) * df['Brent'].pct_change().shift(1)

    # 4. Moving Average (Ci ny bn  lm ng  ln trc)
    df['Price_MA3'] = df['PRICE'].shift(1).rolling(3).mean()
    df['Brent_MA3'] = df['Brent'].shift(1).rolling(3).mean()
    df['USDVND_MA3'] = df['VND_USD'].shift(1).rolling(3).mean()

    df = df.dropna()
    return df

def create_sequences(data, seq_length):
    X, y = [], []
    for i in range(len(data) - seq_length):
        X.append(data[i:i+seq_length])
        y.append(data[i+seq_length, 0])
    return np.array(X), np.array(y)

def evaluate(y_true, y_pred, set_name):
    mse = mean_squared_error(y_true, y_pred)
    rmse = np.sqrt(mse)
    mae = mean_absolute_error(y_true, y_pred)
    r2 = r2_score(y_true, y_pred)
    mape = np.mean(np.abs((y_true - y_pred) / (y_true + 1e-8))) * 100

    if len(y_true) > 1:
        actual_dir = np.sign(np.diff(y_true.flatten()))
        pred_dir = np.sign(np.diff(y_pred.flatten()))
        da = np.mean(actual_dir == pred_dir) * 100
    else:
        da = 0

    print(f"\n{'='*70}")
    print(f"{set_name:^70}")
    print(f"{'='*70}")
    print(f"  RMSE:           {rmse:>15,.2f}")
    print(f"  MAE:            {mae:>15,.2f}")
    print(f"  R Score:       {r2:>15.4f}")
    print(f"  MAPE:           {mape:>14.2f} %")
    print(f"  Dir Accuracy:   {da:>14.2f} %")
    print(f"{'='*70}")

    return {'RMSE': rmse, 'MAE': mae, 'R2': r2, 'MAPE': mape, 'DA': da}
