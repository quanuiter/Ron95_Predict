def create_lag_only_features(df_input):
    df = df_input.copy()

    df['Lag_1'] = df['Price'].shift(1)

    df['Price_Pct'] = df['Price'].pct_change()
    df['Brent_Pct'] = df['Brent'].pct_change()
    df['WTI_Pct'] = df['WTI'].pct_change()
    df['USDVND_Pct'] = df['VND_USD'].pct_change()

    df['Brent_WTI_Spread'] = df['Brent'] - df['WTI']
    df['Brent_WTI_Ratio'] = df['Brent'] / (df['WTI'] + 1e-8)
    df['USD_Brent_Interact'] = df['VND_USD'] * df['Brent_Pct']

    df['Price_MA3'] = df['Price'].rolling(3).mean()
    df['Brent_MA3'] = df['Brent'].rolling(3).mean()
    df['USDVND_MA3'] = df['VND_USD'].rolling(3).mean()

    df = df.dropna()
    return df

df_feat = create_lag_only_features(df).reset_index(drop=True)
