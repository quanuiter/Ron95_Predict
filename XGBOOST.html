def create_lag_only_features(df_input):
    df = df_input.copy()

    # 1. Lag ca price (OK)
    df['Lag_1'] = df['PRICE'].shift(1)


    df['Price_Pct'] = df['PRICE'].pct_change().shift(1)

    df['Brent_Pct'] = df['Brent'].pct_change().shift(1)
    df['WTI_Pct'] = df['WTI'].pct_change().shift(1)
    df['USDVND_Pct'] = df['VND_USD'].pct_change().shift(1)

    # 3. Interaction
    df['Brent_WTI_Spread'] = (df['Brent'] - df['WTI']).shift(1)
    df['Brent_WTI_Ratio'] = (df['Brent'] / (df['WTI'] + 1e-8)).shift(1)

    # Bin tng tc phc tp: USD (hm qua) * Brent% (hm qua)
    df['USD_Brent_Interact'] = df['VND_USD'].shift(1) * df['Brent'].pct_change().shift(1)

    # 4. Moving Average
    df['Price_MA3'] = df['PRICE'].shift(1).rolling(3).mean()
    df['Brent_MA3'] = df['Brent'].shift(1).rolling(3).mean()
    df['USDVND_MA3'] = df['VND_USD'].shift(1).rolling(3).mean()

    df = df.dropna()
    return df

def calculate_metrics(y_true, y_pred):
    rmse = np.sqrt(mean_squared_error(y_true, y_pred))
    mae = mean_absolute_error(y_true, y_pred)
    r2 = r2_score(y_true, y_pred)
    mape = np.mean(np.abs((y_true - y_pred) / y_true)) * 100

    # Directional Accuracy
    y_true_diff = np.diff(y_true)
    y_pred_diff = np.diff(y_pred)
    da = np.mean((y_true_diff * y_pred_diff) > 0) * 100

    return {
        'RMSE': rmse,
        'MAE': mae,
        'R2': r2,
        'MAPE': mape,
        'DA': da
    }
